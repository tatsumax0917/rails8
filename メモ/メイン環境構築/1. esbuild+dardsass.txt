[package.json]
"scripts": {
	"build": "yarn build:js",
	"build:js": "esbuild app/javascript/*.* --bundle --sourcemap --format=esm --outdir=app/assets/builds --public-path=/assets",
	"build:js:watch": "esbuild app/javascript/*.* --bundle --sourcemap --format=esm --outdir=app/assets/builds --public-path=/assets --watch",
},
※どっちでもいいけど、本番用と開発中--watchありとなしを作っておくといいかも（もし動かない場合は作ろう）h

[Gemfile]
gem "dartsass-rails", "~> 0.5.1"

[command]
bundle install

[Procfile]

dartsass: bin/rails dartsass:watch
